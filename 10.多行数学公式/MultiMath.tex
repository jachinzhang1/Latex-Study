\documentclass{ctexart}

\usepackage{amsmath}
\usepackage{amssymb}

\begin{document}
	% gather & gather*环境（可以用\\换行）
	% 带编号
	\begin{gather}
		a+b=b+a\\
		ab ba
	\end{gather}
	% 不带编号
	\begin{gather*}
		3+5=5+3=8\\
		3 \times 5 = 5 \times 3
	\end{gather*}
	% 在gather前用\notag 阻止编号
	\begin{gather}
		3^2 + 4^2 = 5^2 \notag \\
		5^2 + 12^2 = 13^2 \notag \\
		a^2 + b^2 = c^2 \notag 
	\end{gather}

	% align & align*环境对齐
	\begin{align}
		x &=t & x &= \cos t & x &=t \\
		y+1 &= 2t & y &= \sin(t+1) & y &= \sin t
	\end{align}
	\begin{align*}
		x+y & = 1\\
		2x-y & = 2
	\end{align*}

	% split环境（对齐采用align环境的方式，编号在中间）
	\begin{equation}
		\begin{split}
			\cos 2x & = \cos^2 x - \sin^2 x \\
			&= 2\cos^2 x - 1
		\end{split}
	\end{equation}

	% case环境用于生成大括号
	% 每行公式中使用 & 分隔为两部分，
	% 通常表示值和后面的条件
	\begin{equation}
		D(x) = \begin{cases}
			1, & \text{如果 }x \in \mathbb{Q}; \\
			0, & \text{如果 }x \in
			\mathbb{R}\setminus\mathbb{Q}.
			% \in 用于输出【属于】符号
			% \mathbb 用于输出花体字符
			% \setminus 用于输出反斜杠
		\end{cases}
	\end{equation}

	
\end{document}
