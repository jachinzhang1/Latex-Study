	\begin{table}[h]
		\centering
		\caption{成绩单}\label{tab-score}
		\begin{tabular}{l || c | c | c | p{1.5cm}}% | 产生表格竖线（||双竖线）
			%生成表格环境
			%此处生成5列表格（l=左对齐,c=居中对齐,r=右对齐）
			%p可以规定列宽，内容超过宽度自动换行

			姓名 & 语文 & 数学 & 外语 & 备注 \\
			\hline\hline%产生双横线
